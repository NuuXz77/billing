database yang di perlukan untuk keperluan projek billing:

1. table users (
	id_user (primary key)
	profile_id (foreign key ke table profile)
	user_code (unique)
	role (enum('admin','member'))
	email (varchar)
	full_name (varchar)
	password (varchar)
	status (enum('active','suspended','deleted') default 'active')
	last_active (date_time)
	created_at (timestamp)
	updated_at (timestamp)
)

2. table profile (
	id_profile (primary key)
	profile_code (unique)
	foto_profile (varchar)
	phone (varchar)
	alamat (text)
	kota (varchar)
	provinsi (varchar)
	kode_pos (varchar)
	negara (varchar)
	company_name (varchar nullable)
	created_at (timestamp)
	updated_at (timestamp)
)

3. table hosting_plans (
	id_plan (primary key)
	plan_code (unique)
	name_plan (varchar)
	website (varchar)
	ssh (enum('free','no free'))
	price_monthly (int)
	email (varchar)
	storage (int)
	ssl (enum('active','nonactive'))
	database (enum('free','no free'))
	created_at (timestamp)
	updated_at (timestamp)
)

4. table subscriptions (
	id_subscription (primary key)
	subscription_code (unique)
	user_id (foreign key ke table users)
	plan_id (foreign key ke table hosting_plans)
	status (enum('pending','active','suspended','deleted','canceled'))
	start_date (date)
	end_date (date)
	auto_renew (boolean)
	period_billing (enum('monthly','semiannual','yearly','custom'))
	created_at (timestamp)
	updated_at (timestamp)
)

5. table subdomains (
	id_subdomain (primary key)
	subdomain_code (unique)
	subscription_id (foreign key ke table subscriptions)
	user_id (foreign key ke table users)
	subdomain_web (varchar)
	subdomain_server (varchar)
	status (enum('active','suspended','deleted'))
	document_root (varchar)
	created_at (timestamp)
	updated_at (timestamp)
)

6. table invoices (
	id_invoice (primary key)
	invoice_code (unique)
	subscription_id (foreign key ke table subscriptions)
	user_id (foreign key ke table users)
	invoice_number (unique)
	due_date (date)
	status (enum('pending','paid','failed','expired','refunded','canceled'))
	created_at (timestamp)
	updated_at (timestamp)
)

7. table payments (
	id_payment (primary key)
	payment_code (unique)
	invoice_id (foreign key ke table invoices)
	payment_method (varchar)
	amount (int)
	total_payment (int)
	midtrans_order_id (varchar)
	payment_token (varchar)
	paid_at (datetime)
	status (enum('pending','paid','failed','expired'))
	created_at (timestamp)
	updated_at (timestamp)
)

8. table transactions (
	id_transaction (primary key)
	transaction_code (unique)
	invoice_id (foreign key ke table invoices)
	payment_id (foreign key ke table payments)
	payment_status (enum('pending','success','failed','expired'))
	callback_response (json)
	created_at (timestamp)
	updated_at (timestamp)

)
